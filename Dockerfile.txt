# Use official ASP.NET Core runtime image
FROM mcr.microsoft.com/dotnet/aspnet:8.0 AS base
WORKDIR /app
EXPOSE 80

# Use SDK image to build the app
FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build
WORKDIR /src

# Copy csproj and restore dependencies
COPY ["CollegeManagementAPI.csproj", "./"]
RUN dotnet restore "./CollegeManagementAPI.csproj"

# Copy all files and publish
COPY . .
RUN dotnet publish "./CollegeManagementAPI.csproj" -c Release -o /app/publish

# Final image
FROM base AS final
WORKDIR /app
COPY --from=build /app/publish .
ENTRYPOINT ["dotnet", "CollegeManagementAPI.dll"]
